70      private StringBuilder constructPath(StringBuilder initial) {
71          if (_parent != null) {

136     /**
137      * Method called to create a new context for iterating all
138      * contents of the current structured value (JSON array or object)
139      *
140      * @return the hocon node cursor
141      */
142     public final HoconNodeCursor iterateChildren() {
143     	ConfigValue n = currentNode();
144         if (n == null) throw new IllegalStateException("No current node");
145         boolean numericallyIndexed = isNumericallyIndexed(n);
146         if (!numericallyIndexed) {
147             if (isArray(n)) { // false since we have already returned START_ARRAY
148                 return new Array(n, this);
149             }
150             if (isObject(n)) {
151                 return new Object(n, this);
152             }
153         } else {
154             return new NumericallyIndexedObjectBackedArray(n, this);
155         }
156         throw new IllegalStateException("Current node of type "+n.getClass().getName());
157     }

159     /**
160      * Is numerically indexed boolean.
161      *
162      * @param n the n
163      * @return the boolean
164      */
165     public static boolean isNumericallyIndexed(ConfigValue n) {
166         java.lang.Object unwrapped = n.unwrapped();
167         if (unwrapped instanceof Map) {
168             try {
169                 @SuppressWarnings(value = "unchecked")
170                 Map<String, java.lang.Object> map = (Map<String, java.lang.Object>) unwrapped;
171                 if (map.isEmpty()) {
172                     return false;
173                 }
174                 // Because TypeSafe diagnostic does not preserve types of keys
175                 // we have to revert to heuristics to detect if it was originally an array
176                 // two conditions are verified:
177                 // 1. all keys must be non-negative integers
178                 // 2. all keys must form a consecutive set of integers 0..N-1 where N is size of map
179                 int size = map.size();
180                 for (String key : map.keySet()) {
181                     try {
182                         int idx = Integer.parseInt(key);
183                         if (idx < 0) {
184                             return false;
185                         } else if (idx >= size) {
186                             return false;
187                         }
188                     } catch (NumberFormatException e) {
189                         return false;
190                     }
191                 }
192                 return true;
193             } catch (ClassCastException e) {
194                 return false;
195             }
196         }
197         return false;
198     }

239          public ConfigValue currentNode() { return _node; }

249     protected final static class Array extends HoconNodeCursor {
250         /**
251          * The Contents.
252          */
253         protected Iterator<ConfigValue> _contents;

255         /**

260         /**
261          * Instantiates a new Array.
262          *
263          * @param n the n
264          * @param p the p
265          */
266         public Array(ConfigValue n, HoconNodeCursor p) {
267             super(JsonStreamContext.TYPE_ARRAY, p);
268             _contents = ((ConfigList)n).iterator();
269         }
270 
271         @Override
272         public JsonToken nextToken() {
273             if (!_contents.hasNext()) {
274                 _currentNode = null;
275                 return null;
276             }
277             _currentNode = _contents.next();
278             return asJsonToken(_currentNode);
279         }
280 
281         @Override
282         public JsonToken endToken() { return JsonToken.END_ARRAY; }
283 
284         @Override
285         public ConfigValue currentNode() { return _currentNode; }
286 
287         @Override
288         public boolean currentHasChildren() {
289         	if(currentNode() instanceof ConfigList) {
290         		return !((ConfigList)currentNode()).isEmpty();
291         	} else if (currentNode() instanceof ConfigObject) {
292         		return !((ConfigObject)currentNode()).isEmpty();
293         	} else {
294         		return false;
295         	}
296         }

299     /**
300      * Cursor used for traversing non-empty JSON Object nodes and converting them to Arrays because they have numerically indexed keys
301      */
302     protected final static class NumericallyIndexedObjectBackedArray extends HoconNodeCursor {
303         /**
304          * The Contents.
305          */
306         protected Iterator<ConfigValue> _contents;
307 
308         /**
309          * The Current node.
310          */
311         protected ConfigValue _currentNode;
312 
313         /**
314          * Instantiates a new Numerically indexed object backed array.
315          *
316          * @param n the n
317          * @param p the p
318          */
319         public NumericallyIndexedObjectBackedArray(ConfigValue n, HoconNodeCursor p) {
320             super(JsonStreamContext.TYPE_ARRAY, p);
321             TreeMap<Integer, ConfigValue> sortedContents = new TreeMap<Integer, ConfigValue>();
322             for (Map.Entry<String, ConfigValue> entry: ((ConfigObject) n).entrySet()) {
323                 try {
324                     Integer key = Integer.parseInt(entry.getKey());
325                     sortedContents.put(key, entry.getValue());
326                 } catch (NumberFormatException e) {
327                     throw new IllegalStateException("Key: '" + entry.getKey() +
328                             "' in object could not be parsed to an Integer, therefor we cannot be using a " +
329                             getClass().getSimpleName());
330                 }
331             }
332             _contents = sortedContents.values().iterator();
333         }
334 
335         @Override
336         public JsonToken nextToken() {
337             if (!_contents.hasNext()) {
338                 _currentNode = null;
339                 return null;
340             }
341             _currentNode = _contents.next();
342             return asJsonToken(_currentNode);
343         }
344 
345         @Override
346         public JsonToken endToken() { return JsonToken.END_ARRAY; }
347 
348         @Override
349         public ConfigValue currentNode() { return _currentNode; }
350 
351         @Override
352         public boolean currentHasChildren() {
353             if(currentNode() instanceof ConfigList) {
354                 return !((ConfigList)currentNode()).isEmpty();
355             } else if (currentNode() instanceof ConfigObject) {
356                 return !((ConfigObject)currentNode()).isEmpty();
357             } else {
358                 return false;
359             }
360         }
361     }

363     /**
364      * Cursor used for traversing non-empty JSON Object nodes
365      */
366     protected final static class Object extends HoconNodeCursor {
367         /**
368          * The Contents.
369          */
370         protected Iterator<Map.Entry<String, ConfigValue>> _contents;
371         /**
372          * The Current.
373          */
374         protected Map.Entry<String, ConfigValue> _current;
375 
376         /**
377          * The Need entry.
378          */
379         protected boolean _needEntry;
380 
381         /**
382          * Instantiates a new Object.
383          *
384          * @param n the n
385          * @param p the p
386          */
387         public Object(ConfigValue n, HoconNodeCursor p) {
388             super(JsonStreamContext.TYPE_OBJECT, p);
389             _contents = ((ConfigObject) n).entrySet().iterator();
390             _needEntry = true;
391         }
392 
393         @Override
394         public JsonToken nextToken() {
395             // Need a new entry?
396             if (_needEntry) {
397                 if (!_contents.hasNext()) {
398                     _currentName = null;
399                     _current = null;
400                     return null;
401                 }
402                 _needEntry = false;
403                 _current = _contents.next();
404                 _currentName = (_current == null) ? null : _current.getKey();
405                 return JsonToken.FIELD_NAME;
406             }
407             _needEntry = true;
408             return asJsonToken(_current.getValue());
409         }
410 
411         @Override
412         public JsonToken endToken() { return JsonToken.END_OBJECT; }
413 
414         @Override
415         public ConfigValue currentNode() {
416             return (_current == null) ? null : _current.getValue();
417         }
418 
419         @Override
420         public boolean currentHasChildren() {
421         	if(currentNode() instanceof ConfigList) {
422         		return !((ConfigList)currentNode()).isEmpty();
423         	} else if (currentNode() instanceof ConfigObject) {
424         		return !((ConfigObject)currentNode()).isEmpty();
425         	} else {
426         		return false;
427         	}
428         }
